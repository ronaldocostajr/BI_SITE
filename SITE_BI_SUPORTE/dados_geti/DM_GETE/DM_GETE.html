CREATE OR REPLACE PROCEDURE sch_gete_dados.pr_gera_auditoria(tablename text, trc text DEFAULT 'A'::text)
 LANGUAGE plpgsql
AS $procedure$
declare
	X_function 		text;
	X_trigger  		text;
	X_fields_new	text;
	X_fields_old	text;
	X_controle		text;
	x_schema    	text := 'public';
	x_pk			text;
	x_pk_old		text;
begin
	trc = upper(trc);
	tablename = lower(tablename);
	if trc in ('H','?')  or tablename = '?' then
		RAISE EXCEPTION '[X] PARA GERAR TRIGGER DE AUDITORIA E TRUNCATE TABLE, [A] PARA GERAR TRIGGER DE AUDITORIA, [T] PARA GERAR TRIGGER DE TRUNCATE TABLE, [D] PARA DROPAR TODAS AS TRIGGERS DE AUDITORIA, [H OU ?] PARA VER HELP';
	end if;
	if trc not in ('A','T','D','H','X','?') then
		RAISE EXCEPTION '[X] PARA GERAR TRIGGER DE AUDITORIA E TRUNCATE TABLE, [A] PARA GERAR TRIGGER DE AUDITORIA, [T] PARA GERAR TRIGGER DE TRUNCATE TABLE, [D] PARA DROPAR TODAS AS TRIGGERS DE AUDITORIA, [H OU ?] PARA VER HELP';
	end if;
	if lower(tablename) = 'tb_auditoria' then
		RAISE EXCEPTION 'USUÁRIO NÃO TEM PERMISSÃO PARA ALTERAR A TABELA DE AUDITORIA.';
	end if;
	select lower(column_name)	
	into X_fields_new
	from information_schema.columns
	where table_schema = X_schema
	and   lower(table_name) = lower(tablename)
	limit 1;
	if not found then
		RAISE EXCEPTION 'TABELA NÃO LOCALIZADA.';
	end if;
	begin
		SELECT STRING_AGG(''''||a.attname||'='||''''||'||'||'new.'||a.attname, '||'||''''||' and '||''''||'||' ORDER BY a.attnum)
		into x_pk
		FROM   pg_index i
		JOIN   pg_attribute a ON a.attrelid = i.indrelid AND a.attnum = ANY(i.indkey)
		WHERE  i.indrelid = lower(tablename)::regclass
		AND    i.indisprimary;					
	exception when others then
		x_pk := '';
	end;	
     if x_pk is null then
     	x_pk := ''''||'NO PRIMARY KEY'||'''';
     end if;
	x_pk_old := replace(x_pk, 'new','old');
	If trc in ('A','X') Then	
		begin
			X_function := 'drop function '|| X_schema ||'.fn_aud_'||lower(tablename)||' cascade;';
			execute X_function;
		exception when others then
			null;
		end;
		select string_agg(''''||'"'||lower(column_name)||'": "'||''''||' ||coalesce(NEW.'||lower(column_name)||'::text,'||''''||'NULL'||''''||')'||'||'||''''||'"'||'''' ||'||'||''''||','||''''||'||','' order by ordinal_position) as gr	
		into X_fields_new
		from information_schema.columns
		where table_schema = X_schema
		and   lower(table_name) = lower(tablename)
		group by table_name;
		if not found then
			RAISE EXCEPTION 'TABELA NÃO LOCALIZADA.';
		end if;	
		X_fields_new := ''''||'{'||''''||'||'||substring(X_fields_new,1, length(X_fields_new)-7)||'||'||''''||'}'||'''';
		select string_agg(''''||'"'||lower(column_name)||'": "'||''''||' ||coalesce(OLD.'||lower(column_name)||'::text,'||''''||'NULL'||''''||')'||'||'||''''||'"'||'''' ||'||'||''''||','||''''||'||','' order by ordinal_position) as gr
		into X_fields_old
		from information_schema.columns
		where table_schema = X_schema
		and   lower(table_name) = lower(tablename)
		group by table_name;
		X_fields_old := ''''||'{'||''''||'||'||substring(X_fields_old,1, length(X_fields_old)-7)||'||'||''''||'}'||'''';
		X_function := 'create or replace function '|| X_schema ||'.fn_aud_'|| tablename ||'() 
			returns trigger as 
		$$ 
		declare 
			x_fl_insert char(01);
			x_fl_update char(01);
			x_fl_delete char(01);
			x_user      text;
		begin 
		     begin
				select fl_insert, fl_update, fl_delete
				into  x_fl_insert, x_fl_update, x_fl_delete
				from  '|| X_schema ||'.tb_auditoria_controle au
				where lower(au.nm_tabela) = '||''''||lower(tablename)||''''||'
				order by au.dt_movimento desc
				limit 1;
				if not found then
					x_fl_insert := '||''''||'S'||''''||';
					x_fl_update := '||''''||'S'||''''||';
					x_fl_delete := '||''''||'S'||''''||';
				end if;
			exception when others then
				x_fl_insert := '||''''||'S'||''''||';
				x_fl_update := '||''''||'S'||''''||';
				x_fl_delete := '||''''||'S'||''''||';
			end;
			if (TG_OP = '||''''||'INSERT'||''''||') then
				if x_fl_insert = '||''''||'S'||''''||' then
					begin
						x_user := coalesce(new.nm_usuario,USER);
					exception when others then
						x_user := USER;
					end;
					begin
						insert into '|| X_schema ||'.tb_auditoria(nm_tabela, ds_movimento, ds_error, dt_movimento, nm_usuario, nu_ip, ds_texto, pk_tabela)
						values('||''''||lower(tablename)||''''||','||''''||'INS'||''''||','||''''||'DML'||''''||','||'now()'||','||'x_user'||','||'inet_client_addr()'||','||X_fields_new||','||x_pk||');
					exception when others then
						null;
					end;
				end if;
				return NEW;
			elsif (TG_OP = '||''''||'UPDATE'||''''||') then
				if x_fl_update = '||''''||'S'||''''||' then
					begin
						x_user := coalesce(new.nm_usuario,USER);
					exception when others then
						x_user := USER;
					end;
					begin
						insert into '|| X_schema ||'.tb_auditoria(nm_tabela, ds_movimento, ds_error, dt_movimento, nm_usuario, nu_ip, ds_texto, pk_tabela)
						values('||''''||lower(tablename)||''''||','||''''||'UPD_OLD'||''''||','||''''||'DML'||''''||','||'now()'||','||'x_user'||','||'inet_client_addr()'||','||X_fields_old||','||x_pk||');
						insert into '|| X_schema ||'.tb_auditoria(nm_tabela, ds_movimento, ds_error, dt_movimento, nm_usuario, nu_ip, ds_texto, pk_tabela)
						values('||''''||lower(tablename)||''''||','||''''||'UPD_NEW'||''''||','||''''||'DML'||''''||','||'now()'||','||'x_user'||','||'inet_client_addr()'||','||X_fields_new||','||x_pk||');
					exception when others then
						null;
					end;
				end if;
				return NEW;
			elsif (TG_OP = '||''''||'DELETE'||''''||') then
				IF x_fl_delete = '||''''||'S'||''''||' then
					begin
						x_user := coalesce(old.nm_usuario,USER);
					exception when others then
						x_user := USER;
					end;
					begin
						insert into '|| X_schema ||'.tb_auditoria(nm_tabela, ds_movimento, ds_error, dt_movimento, nm_usuario, nu_ip, ds_texto, pk_tabela)
						values('||''''||lower(tablename)||''''||','||''''||'DEL'||''''||','||''''||'DML'||''''||','||'now()'||','||'x_user'||','||'inet_client_addr()'||','||X_fields_old||','||x_pk_old||');
					exception when others then
						null;
					end;
				end if;
				return OLD;
			end if;
		end;
		$$
		language plpgsql;';
		execute X_function;
		x_trigger := 'create or replace trigger tg_aud_'||tablename||'
					  after insert or update or delete on '||tablename||'
					  for each row
					  execute procedure fn_aud_'|| tablename ||'();';
		execute x_trigger;
		insert into tb_auditoria
		(nm_tabela, ds_movimento, ds_error, dt_movimento, nm_usuario, nu_ip, ds_texto)
		values (lower(tablename), 'TRIGGER', 'TRG.A.001'::character varying, now(), USER, inet_client_addr(), '{"Ação":"Trigger de auditoria de Registros gerada para a tabela: ['|| lower(tablename)||']"}');
	end if;
	if  trc in ('T','X') then
		begin
			X_function := 'drop function '|| X_schema ||'.fn_audt_'||lower(tablename)||' cascade;';
			execute X_function;
		exception when others then
			null;
		end;
		X_function := 'create or replace function '|| X_schema ||'.fn_audt_'|| tablename ||'() 
			returns trigger as 
		$$ 
		declare 
			X_function	text;
			x_fl_truncate  char(01) := '||''''||'N'||''''||';
		begin 
			begin
				select fl_truncate
				into  x_fl_truncate
				from  '|| X_schema ||'.tb_auditoria_controle au
				where upper(au.nm_tabela) = '||''''||upper(tablename)||''''||'
				order by au.dt_movimento desc
				limit 1;
				if not found then
					RAISE EXCEPTION '||''''||'TRUNCATE NÃO PERMITIDO'||''''||';
				end if;
			exception when others then
				RAISE EXCEPTION '||''''||'TRUNCATE NÃO PERMITIDO'||''''||';
			end;
			if x_fl_truncate = '||''''||'N'||''''||' then
				RAISE EXCEPTION '||''''||'TRUNCATE NÃO PERMITIDO'||''''||';
			end if;
			if (TG_OP = '||''''||'TRUNCATE'||''''||') then
				begin
					insert into '|| X_schema ||'.tb_auditoria(nm_tabela, ds_movimento, ds_error, dt_movimento, nm_usuario, nu_ip, ds_texto)
					values('||''''||upper(tablename)||''''||','||''''||'TRC'||''''||','||''''||'DML'||''''||','||'now()'||','||'USER'||','||'inet_client_addr()'||','|| ''''||'TRUNCATE TABLE'||'''' ||');
				exception when others then
					null;
				end;
				return NEW;
			end if;
		end;
		$$
		language plpgsql;';	
		execute X_function;	
		x_trigger := 'create or replace trigger tg_audt_'||tablename||'
					  after truncate on '||tablename||'
					  for statement
					  execute procedure fn_audt_'|| tablename ||'();';
		execute x_trigger;
		insert into tb_auditoria
		(nm_tabela, ds_movimento, ds_error, dt_movimento, nm_usuario, nu_ip, ds_texto)
		values(upper(tablename), 'TRIGGER', 'TRG.A.002'::character varying, now(), USER, inet_client_addr(), '{"Ação":"Trigger de auditoria de Truncate gerada para a tabela: ['|| lower(tablename)||']"}');
	end if;
	if trc = 'D' then
		begin
			X_function := 'drop function '|| X_schema ||'.fn_aud_'||lower(tablename)||' cascade;';
			execute X_function;
			insert into X_schema.tb_auditoria
			(nm_tabela, ds_movimento, ds_error, dt_movimento, nm_usuario, nu_ip, ds_texto)
			values(upper(tablename), 'TRIGGER', 'TRG.R.003'::character varying, now(), USER, inet_client_addr(), '{"Ação":"Trigger de auditoria de Registros removida da tabela: ['|| lower(tablename)||']"}');
		exception when others then
			null;
		end;
		begin
			X_function := 'drop function '|| X_schema ||'.fn_audt_'||lower(tablename)||' cascade;';
			execute X_function;
			insert into X_schema.tb_auditoria
			(nm_tabela, ds_movimento, ds_error, dt_movimento, nm_usuario, nu_ip, ds_texto)
			values(upper(tablename), 'TRIGGER', 'TRG.R.004'::character varying, now(), USER, inet_client_addr(), '{"Ação":"Trigger de auditoria de Truncate removida da tabela: ['|| lower(tablename)||']"}');
		exception when others then
			null;
		end;
	end if;
end
