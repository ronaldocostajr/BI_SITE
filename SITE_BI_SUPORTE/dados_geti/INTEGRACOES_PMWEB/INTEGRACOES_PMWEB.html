View VW_PMW_RESERVA




	

		Query



	
CREATE VIEW ${nameWithSchemaName} AS SELECT
	R.ID								AS ID_RESERVA,
	R.NM_CONTATO						AS NM_CONTATO,
	TRIM(R.NR_DDD||' '||R.NR_TELEFONE)	AS NU_FONE,
	R.NM_EMAIL							AS DS_EMAIL,
	U.NM_PESSOA 						AS NM_UNIDADE,
	RT.DS_TIPO							AS TP_RESERVA,
	N.SG_NEGOCIO						AS DS_NEGOCIO,
	UC.DS_CATEGORIA						AS NM_CATEGORIA,
	AUX.NO_PRODUTO						AS NM_PRODUTO,
	RS.DS_STATUS						AS ST_RESERVA,
	(SELECT SUM(V.VL_PAGAMENTO * V.VL_CAMBIO) FROM DATALAKE.TB_INFO_RESERVA_VALOR V WHERE V.RESERVA_ID = R.ID)	AS VL_RESERVA,
	R.DT_INCLUSAO,
	AUX.DT_UTILIZACAO,
	US.NM_LOGIN							AS NM_USUARIO,
	R.DT_INGESTAO						AS DT_INGESTAO,
	R.DT_UPDATE 
FROM
	DATALAKE.TB_INFO_RESERVA R
	INNER JOIN DATALAKE.TB_INFO_PESSOA U			ON U.ID				= R.UNIDADE_ID
	INNER JOIN DATALAKE.TB_INFO_RESERVA_STATUS RS	ON RS.ID			= R.RESERVA_STATUS_ID
	LEFT JOIN DATALAKE.TB_INFO_RESERVA_TIPO RT		ON RT.ID			= R.RESERVA_TIPO_ID
	LEFT JOIN DATALAKE.TB_INFO_USUARIO US			ON US.ID			= R.USUARIO_ID
	LEFT JOIN DATALAKE.TB_INFO_NEGOCIO N			ON N.ID				= R.NEGOCIO_ID
	LEFT JOIN DATALAKE.TB_INFO_UNIDADE_CATEGORIA UC	ON UC.ID			= R.UNIDADE_CATEGORIA_ID
	LEFT JOIN
		(SELECT
			X.NU_RESERVA														AS CO_RESERVA,
			MIN(X.DT_UTILIZACAO)												AS DT_UTILIZACAO,
			LISTAGG(X.NO_PRODUTO, ' | ') WITHIN GROUP (ORDER BY X.NO_PRODUTO)	AS NO_PRODUTO
		FROM
			(SELECT 
				I.RESERVA_ID				AS NU_RESERVA,
				P.NM_PESSOA					AS NO_PRODUTO,
				MIN(I.DT_UTILIZACAO_INI)	AS DT_UTILIZACAO
			FROM
				DATALAKE.TB_INFO_RESERVA_INSUMO I
				INNER JOIN DATALAKE.TB_INFO_PESSOA P		ON P.ID	= I.FORNECEDOR_ID
			WHERE I.RESERVA_ID IS NOT NULL
			GROUP BY
				I.RESERVA_ID,
				P.NM_PESSOA	
			) X
		GROUP BY X.NU_RESERVA
		) AUX										ON AUX.CO_RESERVA	= R.ID
WHERE R.RESERVA_STATUS_ID = 12
