View V_HOTRANSACOES
Query
 
CREATE VIEW ${nameWithSchemaName} AS SELECT 
	CASE
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='D' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL<0 THEN 1
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='C' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL<0 THEN 1
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='D' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL>0 THEN 1
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='C' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL>0 THEN 1
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='D' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL<0 THEN 0 
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='C' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL<0 THEN 0
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='D' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL>0 THEN 0
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='C' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL>0 THEN 0
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='D' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL>0 THEN 0
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='C' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL>0 THEN 0	
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='D' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL<0 THEN 0
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='C' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL<0 THEN 0
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='D' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL>0 THEN 1
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='C' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL>0 THEN 1	
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='D' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL<0 THEN 1
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='C' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL<0 THEN 1
		ELSE 0
	END AS contabilizacaoInversa,
	trunc(T.DATATRANSACAO) AS DATALANCAMENTO,
	'OPERA' AS sistemaOrigem,
	t.hotel,
	t.CODIGO,t.DESCRICAO,
	sum(ABS(VALORCONTABIL)) AS VALORCONTABIL,
	dc.dc AS creditodebito,
	t.FLGAJUSTE
FROM 
	(SELECT 'D' AS dc FROM DUAL UNION ALL SELECT 'C' FROM DUAL) DC,
	HO.TRANSACOESFINANCEIRAS t
	JOIN HO.POSICAOAUDITORIA PA ON T.HOTEL =PA.HOTEL AND T.DATALANCAMENTO <= PA.ULTIMAAUDITORIA 
WHERE t.STATUSLANCTO IS NULL
GROUP BY 
	CASE
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='D' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL<0 THEN 1
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='C' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL<0 THEN 1
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='D' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL>0 THEN 1
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='C' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL>0 THEN 1
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='D' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL<0 THEN 0
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='C' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL<0 THEN 0
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='D' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL>0 THEN 0
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='C' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL>0 THEN 0
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='D' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL>0 THEN 0
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='C' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL>0 THEN 0	
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='D' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL<0 THEN 0
		WHEN t.TRANSACTIONTYPE='R' AND dc.dc='C' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL<0 THEN 0
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='D' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL>0 THEN 1
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='C' AND t.FLGAJUSTE='N' AND t.VALORCONTABIL>0 THEN 1	
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='D' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL<0 THEN 1
		WHEN t.TRANSACTIONTYPE='P' AND dc.dc='C' AND t.FLGAJUSTE='S' AND t.VALORCONTABIL<0 THEN 1
		ELSE 0		
	END,
	trunc(t.DATATRANSACAO),

	t.FLGAJUSTE
