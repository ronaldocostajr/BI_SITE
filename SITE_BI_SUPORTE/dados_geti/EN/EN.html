View VW_CONSUMO_TESTE
Query
 
CREATE VIEW ${nameWithSchemaName} AS SELECT 	UPPER(C.LOJA) AS LOJA, CLI.IDCLIENTE, 
REPLACE(TRIM(UPPER(REGEXP_REPLACE(CLI.CLIENTE,'[0-9]',''))),CHR(10),'') AS CLIENTE, 
trunc(C.DATAMOVIMENTO) AS datavenda , upper(C.SUBCATEGORIA1) AS SUBCATEGORIA1 , 
UPPER(C.SUBCATEGORIA2) AS SUBCATEGORIA2 ,UPPER(C.DESCRICAOPRODUTO) AS DESCRICAOPRODUTO, 
TO_CHAR(C.VALORTOTAL,'FM999G999G999D99') AS VALORTOTAL, 
C.QUANTIDADE
FROM EN.VENDASPDVEN C,
