PR_MQ_GARBAGE_COLLECTOR Standalone procedure or function 
CREATE OR REPLACE PROCEDURE         PR_MQ_GARBAGE_COLLECTOR AS
BEGIN
DECLARE

	CURSOR C_QUEUE_RETIRED IS
		SELECT QR.SQ_QUEUE_RETIRED, QR.SQ_QUEUE_MESSAGE , QR.SQ_STATUS_QUEUE, QR.NM_USER_RETIRED , QR.DS_OBSERVACAO_RETIRED , SQ.FL_ARQUIVA , SQ.FL_STATUS 
		FROM AVIVAMQ.TB_MQ_QUEUE_RETIRED QR
		JOIN AVIVAMQ.TB_MQ_STATUS_QUEUE SQ ON SQ.SQ_STATUS_QUEUE = QR.SQ_STATUS_QUEUE 
		JOIN AVIVAMQ.TB_MQ_QUEUE MQ ON MQ.SQ_QUEUE = QR.SQ_QUEUE 
		WHERE QR.SQ_STATUS_RETIRED 	= 0
		AND   MQ.FL_PERSISTENTE 	= 'N';
	
	CURSOR C_GARBAGE_USED IS 
		SELECT MQ.*
		FROM AVIVAMQ.TB_MQ_QUEUE_MESSAGE MQ
		JOIN AVIVAMQ.TB_MQ_STATUS_QUEUE SQ 	ON SQ.SQ_STATUS_QUEUE = MQ.SQ_STATUS_QUEUE 
		JOIN AVIVAMQ.TB_MQ_QUEUE QU 		ON QU.SQ_QUEUE = MQ.SQ_QUEUE 
		WHERE SQ.FL_STATUS 	= 'P';
	
	CURSOR C_GARBAGE_RETIRED IS 
		SELECT MQ.*
		FROM AVIVAMQ.TB_MQ_QUEUE_MESSAGE MQ
		JOIN AVIVAMQ.TB_MQ_STATUS_QUEUE SQ 	ON SQ.SQ_STATUS_QUEUE = MQ.SQ_STATUS_QUEUE 
		JOIN AVIVAMQ.TB_MQ_QUEUE QU 		ON QU.SQ_QUEUE = MQ.SQ_QUEUE 
		WHERE SQ.FL_STATUS 	= 'E'
		AND   SQ.FL_ARQUIVA = 'S';	

	CURSOR C_GARBAGE_REPROCESSAMENTO IS 
		SELECT MQ.*
		FROM AVIVAMQ.TB_MQ_QUEUE_MESSAGE MQ
		JOIN AVIVAMQ.TB_MQ_STATUS_QUEUE SQ 	ON SQ.SQ_STATUS_QUEUE = MQ.SQ_STATUS_QUEUE 
		JOIN AVIVAMQ.TB_MQ_QUEUE QU 		ON QU.SQ_QUEUE = MQ.SQ_QUEUE 
		WHERE SQ.FL_STATUS 		= 'E'
		AND   SQ.FL_REPROCESSA 	= 'S'
		AND   SQ.FL_ARQUIVA 	= 'N';
	
	XQUEUE	AVIVAMQ.TB_MQ_STATUS_QUEUE%ROWTYPE;
		
BEGIN
	FOR C1 IN C_QUEUE_RETIRED LOOP
		IF C1.FL_STATUS <> 'E' AND C1.FL_ARQUIVA = 'N' THEN
			UPDATE AVIVAMQ.TB_MQ_QUEUE_RETIRED QR 
			SET    QR.SQ_STATUS_RETIRED = 10,
				   QR.DS_OBSERVACAO_RETIRED = 'NÃO RETIRADO, STATUS NÃO IDENTIFICADO COMO ARQUIVAMENTO: '|| NVL(QR.DS_OBSERVACAO_RETIRED, ' ')
			WHERE  QR.SQ_QUEUE_RETIRED  = C1.SQ_QUEUE_RETIRED;
			CONTINUE;
		END IF;
	
		SELECT SQ.FL_STATUS 
		INTO   XQUEUE.FL_STATUS
		FROM   AVIVAMQ.TB_MQ_QUEUE_MESSAGE QM
		JOIN   AVIVAMQ.TB_MQ_STATUS_QUEUE  SQ ON SQ.SQ_STATUS_QUEUE = QM.SQ_STATUS_QUEUE 
		WHERE  QM.SQ_QUEUE_MESSAGE = C1.SQ_QUEUE_MESSAGE;
	
		IF XQUEUE.FL_STATUS IN ('I','P') THEN
			UPDATE AVIVAMQ.TB_MQ_QUEUE_RETIRED QR 
			SET    QR.SQ_STATUS_RETIRED = 10,
				   QR.DS_OBSERVACAO_RETIRED = 'NÃO RETIRADO, MENSAGEM COM STATUS INICIAL OU PROCESSADA'
			WHERE  QR.SQ_QUEUE_RETIRED  = C1.SQ_QUEUE_RETIRED;
			CONTINUE;
		END IF;
	
		UPDATE AVIVAMQ.TB_MQ_QUEUE_MESSAGE MQ
		SET    MQ.SQ_STATUS_QUEUE = C1.SQ_STATUS_QUEUE,
			   MQ.DS_OBSERVACAO   = 'RETIRADO PELO OPERADOR: '||C1.NM_USER_RETIRED||' - MOTIVO: '||C1.DS_OBSERVACAO_RETIRED
		WHERE  MQ.SQ_QUEUE_MESSAGE = C1.SQ_QUEUE_MESSAGE;
	
		UPDATE AVIVAMQ.TB_MQ_QUEUE_RETIRED QR 
		SET    QR.SQ_STATUS_RETIRED = 1
		WHERE  QR.SQ_QUEUE_RETIRED  = C1.SQ_QUEUE_RETIRED;	
	END LOOP;	
	
	FOR C1 IN C_GARBAGE_USED LOOP 
		BEGIN
			INSERT INTO AVIVAMQ.TB_MQ_QUEUE_MESSAGE_USED (SQ_QUEUE_MESSAGE, SQ_QUEUE, SQ_STATUS_QUEUE, SQ_TABELA_ORIGEM, DS_ROWID, DT_ENTREGA_QUEUE, DT_CAPTURA_QUEUE, QT_PROCESSAMENTO, DS_MENSAGEM, DS_OBSERVACAO, SQ_ROWID_DESTINO) 
		    VALUES (C1.SQ_QUEUE_MESSAGE, C1.SQ_QUEUE, C1.SQ_STATUS_QUEUE, C1.SQ_TABELA_ORIGEM, C1.DS_ROWID, C1.DT_ENTREGA_QUEUE, C1.DT_CAPTURA_QUEUE, C1.QT_PROCESSAMENTO, C1.DS_MENSAGEM, C1.DS_OBSERVACAO, C1.SQ_ROWID_DESTINO);
	
		   	DELETE FROM AVIVAMQ.TB_MQ_QUEUE_MESSAGE MQ
		   	WHERE  MQ.SQ_QUEUE_MESSAGE = C1.SQ_QUEUE_MESSAGE;
		EXCEPTION WHEN OTHERS THEN
			NULL;
		END;
	END LOOP;

	FOR C1 IN C_GARBAGE_RETIRED LOOP 
		BEGIN
			INSERT INTO AVIVAMQ.TB_MQ_QUEUE_MESSAGE_RETIRED (SQ_QUEUE_MESSAGE, SQ_QUEUE, SQ_STATUS_QUEUE, SQ_TABELA_ORIGEM, DS_ROWID, DT_ENTREGA_QUEUE, DT_CAPTURA_QUEUE, QT_PROCESSAMENTO, DS_MENSAGEM, DS_OBSERVACAO, SQ_ROWID_DESTINO) 
		    VALUES (C1.SQ_QUEUE_MESSAGE, C1.SQ_QUEUE, C1.SQ_STATUS_QUEUE, C1.SQ_TABELA_ORIGEM, C1.DS_ROWID, C1.DT_ENTREGA_QUEUE, C1.DT_CAPTURA_QUEUE, C1.QT_PROCESSAMENTO, C1.DS_MENSAGEM, C1.DS_OBSERVACAO, C1.SQ_ROWID_DESTINO);
	
		   	DELETE FROM AVIVAMQ.TB_MQ_QUEUE_MESSAGE MQ
		   	WHERE  MQ.SQ_QUEUE_MESSAGE = C1.SQ_QUEUE_MESSAGE;
		EXCEPTION WHEN OTHERS THEN
			NULL;
		END;
	END LOOP;

	FOR C1 IN C_GARBAGE_REPROCESSAMENTO LOOP 
		UPDATE 	AVIVAMQ.TB_MQ_QUEUE_MESSAGE Q
		SET 	Q.SQ_STATUS_QUEUE 		= 0
		WHERE 	Q.SQ_QUEUE_MESSAGE 		= C1.SQ_QUEUE_MESSAGE;
	END LOOP;
END;
END;
